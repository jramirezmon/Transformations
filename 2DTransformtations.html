int posx = 0;

float angle = 0;

void keyReleased() {
	angle = angle + radians(45);
  }
  
void pintarCuadrado(){
  fill(50,100,40);
  push();
  posx = mouseX;
  posy = mouseY;
  translate(posx,posy);
  rotate(angle);

  pop();
}
 
  
void setup(){
  size(300,300,P2D);

//Triángulo Grande 1:
float angleT = 3.1415/4;
int posxT = 100;
int posyT = 100;

//Paralelogramo:
float angleP = 0;
int posxP = 300;
int posyP = 80;

void keyReleased() {
	if((dist(posxT,posyT,mouseX,mouseY)<20) && (keyPressed)){
	  if(key=='a'){
		angleT = angleT + radians(45);
		delay(100);
	  }
	if(key=='s'){
	  angleT = angleT - radians(45);
	  delay(100);
	}
  }
  if((dist(posxP,posyP,mouseX,mouseY)<20) && (keyPressed)){
	if(key=='a'){
	angleP = angleP + radians(45);
	delay(100);
	}
	if(key=='s'){
	angleP = angleP - radians(45);
	delay(100);
	} 
  }
  }
  
void pintarTriangulo(){
  keyReleased();
  if((dist(posxT,posyT,mouseX,mouseY)<20) && (mousePressed)){  
  posxT = mouseX;
  posyT = mouseY;

  }
  fill(210,50,60);
  push();
  translate(posxT,posyT);
  rotate(angleT);
  triangle(-30,-30,70,-30,-30,70);
  fill(200,45,65);
  noStroke();
  circle(0,0,20);
  pop();
}

//Programación paralelogramo
void pintarParalelogramo(){
  
  if((dist(posxP,posyP,mouseX,mouseY)<20) && (mousePressed)){  
  posxP = mouseX;
  posyP = mouseY;

  }
  fill(80,20,170);
  push();
  translate(posxP,posyP);
  rotate(angleP);
  quad(-50,-25,0,-25,50,25,0,25);
  fill(70,20,190);
  noStroke();
  circle(0,0,20);
  pop();
}
  
void setup(){ 
  size(400,300,P2D);
}

void draw(){  
  background(215,120,63);
  pintarTriangulo();
  pintarParalelogramo();
}
